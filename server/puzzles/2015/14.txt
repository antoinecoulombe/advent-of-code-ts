const totals = [];

const time = 2503;

deers.forEach(dear => {
	const totalLength = dear.duration + dear.rest;
  const totalMoves = time / totalLength;
  const flooredMoves = Math.floor(totalMoves);
  const diffMoves = Math.round((totalMoves - flooredMoves) * totalLength); 
  let totalDistance = Math.floor(totalMoves) * dear.speed * dear.duration;
  console.log(totalLength, totalMoves, flooredMoves, diffMoves, totalDistance);
  if (diffMoves > 0) {
  	const additionalMoves = diffMoves < dear.duration ? diffMoves : dear.duration;
    totalDistance += additionalMoves * dear.speed;
  }
  totals.push({name: dear.name, distance: totalDistance});
});

const maxDistance = Math.max(...totals.map(total => total.distance));
const furthest = totals.find(total => total.distance === maxDistance);
